leetcode451.根据字符出现频率排序

2023-09-12
630- 2m

题目

给定一个字符串 s ，根据字符出现的 频率 对其进行 降序排序 。一个字符出现的 频率 是它出现在字符串中的次数。

返回 已排序的字符串 。如果有多个答案，返回其中任何一个。

示例 1:

输入: s = "tree"
输出: "eert"
解释: 'e'出现两次，'r'和't'都只出现一次。
因此'e'必须出现在'r'和't'之前。此外，"eetr"也是一个有效的答案。

示例 2:

输入: s = "cccaaa"
输出: "cccaaa"
解释: 'c'和'a'都出现三次。此外，"aaaccc"也是有效的答案。
注意"cacaca"是不正确的，因为相同的字母必须放在一起。

示例 3:

输入: s = "Aabb"
输出: "bbAa"
解释: 此外，"bbaA"也是一个有效的答案，但"Aabb"是不正确的。
注意'A'和'a'被认为是两种不同的字符。

提示:

  • 1 <= s.length <= 5 * 105
  • s 由大小写英文字母和数字组成

解答

思路：首先使用hashmap对字符串中的字符进行计数，然后使用vector数组将hashmap的元素进行拷贝，将vector以计数的从大到小进行排序。最后对排序后的vector进行遍历，得到最终的字符串，C++代码实现如下：

class Solution {
public:
    string frequencySort(string s) {
        unordered_map<char, int> s_count_map;
        for (char c:s) {
            s_count_map[c]++;
        }
        // 复制
        vector<pair<char, int>> s_count_vector(s_count_map.begin(), s_count_map.end());
        //从大到小排序
        sort(s_count_vector.begin(), s_count_vector.end(), [](pair<char, int> a, pair<char, int> b){return a.second > b.second;});
        string result;
        for(auto cnt:s_count_vector) {
            int cnt_second = cnt.second;
            while (cnt_second--) {
                result+=cnt.first;
            }
        }
        return result;
    }
};
  • 一些细节优化后的代码（成绩没变）
class Solution {
public:
    string frequencySort(string s) {
        unordered_map<char, int> s_count_map;

        // 统计字符频率
        for (char c : s) {
            s_count_map[c]++;
        }

        // 构建结果字符串
        string result;
        result.reserve(s.size()); // 预分配足够的内存

        // 构建频率字符对，并按频率降序排序
        vector<pair<char, int>> s_count_vector;
        for (const auto& kv : s_count_map) {
            s_count_vector.emplace_back(kv);
        }
        sort(s_count_vector.begin(), s_count_vector.end(), [](pair<char, int> a, pair<char, int> b){ return a.second > b.second; });

        // 构建结果字符串
        for (const auto& cnt : s_count_vector) {
            result.append(cnt.second, cnt.first); // 使用append方法
        }

        return result;
    }
};
