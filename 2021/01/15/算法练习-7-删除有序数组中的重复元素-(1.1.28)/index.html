1.1.28 Remove duplicates. Modify the test client in BinarySearch to remove any duplicate keys in the whitelist after the sort.

（这是一道Leetcode题目，原题描述如下：

Remove Duplicates from Sorted Array
Given a sorted array, remove the duplicates in place such that each element appear only once and return the new length.
Do not allocate extra space for another array, you must do this in place with constant memory.
For example,
Given input array nums = [1,1,2],
Your function should return length = 2, with the first two elements of nums being 1 and 2 respectively. It doesn't matter what you leave beyond the new length.

  • 我自己的思路

    写一个接收数组信息的函数getArr，输入数组大小和数组内容后，使用冒泡排序进行数组的排序，从小到大，方便之后的去重。

    又写了一个生成数组元素个数的count数组的函数getCount，返回一个数组count，count[i]表示数组arr中i元素的数量。

    然后写了一个函数delSurplus，参数是两个数组（arr和count），通过for循环将多余的元素用后面的元素依次覆盖，但是我的方法有个缺点是不支持数组里负数的元素，可以加另一个count进行去重。

package top.leafii.practice.charpter01;

import edu.princeton.cs.algs4.StdIn;
import edu.princeton.cs.algs4.StdOut;

import java.util.Arrays;

public class BinarySearch {


    private BinarySearch() { }

    /**
     * 获取key在数组a中的位置，若没有则返回-1
     * @param a
     * @param key
     * @return
     */
    public static int indexOf(int[] a, int key) {
        int lo = 0;
        int hi = a.length - 1;
        while (lo <= hi) {
            // Key is in a[lo..hi] or not present.
            int mid = lo + (hi - lo) / 2;
            if      (key < a[mid]) hi = mid - 1;
            else if (key > a[mid]) lo = mid + 1;
            else return mid;
        }
        return -1;
    }

    /**
     * 获取key在数组a中的位置，若没有则返回-1
     * @param key
     * @param a
     * @return
     */
    public static int rank(int key, int[] a) {
        return indexOf(a, key);
    }

    /**
     * 对数组进行冒泡排序
     * @param arr
     */
    public static void bubbleSort(int arr[]) {

        for(int i =0 ; i<arr.length-1 ; i++) {

            for(int j=0 ; j<arr.length-1-i ; j++) {

                if(arr[j]>arr[j+1]) {
                    int temp = arr[j];

                    arr[j]=arr[j+1];

                    arr[j+1]=temp;
                }
            }
        }
    }

    /**
     * 生成手动输入后的有序数组
     * @return
     */
    public static int[] getArr(){
        StdOut.print("请输入数组的大小（一个整数）:");
        int size = StdIn.readInt();
        int[] arr = new int[size];
        StdOut.println("请输入数组的数字，以空格隔开或者回车隔开:");
        for (int i = 0; i < size; i++) {
            arr[i] = StdIn.readInt();
        }
        bubbleSort(arr);
        StdOut.print("将输入的数组排序后:");
        for (int i = 0; i < size; i++) {
            StdOut.print("arr["+ i + "]=" + arr[i] + "\t");
        }
        StdOut.println();
        return arr;
    }

    /**
     * 生成计数数组
     * @param arr
     */
    public static int[] getCount(int arr[]){
        int[] count = new int[arr[arr.length-1]+1];
        for (int i = 0; i < count.length; i++) {
            count[i] = 0;
        }
        for (int i = 0; i < arr.length; i++) {
            count[arr[i]]++;
        }
        StdOut.print("计数数组的值为：");
        for (int i = 0; i < count.length; i++) {
            System.out.print("count["+i+"]="+count[i]+" ");
        }
        StdOut.println();
        return count;
    }

    /**
     * 通过count数组删除arr中的重复元素，并返回一个新的数组
     * @param arr
     * @param count
     * @return
     */
    public static int[] delSurplus(int[] arr,int[] count){
        int del = 0;
        int size = arr.length;
        for (int i = 0; i < count.length; i++) {
            while(count[i]>=2){
                del = rank(i,arr);
                for (int j = del; j < arr.length-1; j++) {
                    arr[j] = arr[j+1];

                }
                count[i]--;
                size--;
            }
        }
        int[] array = new int[size];
        for (int i = 0; i < array.length; i++) {
            array[i] = arr[i];
        }
        return array;
    }
    public static void main(String[] args) {
        int[] a = getArr();
        int[] count = getCount(a);
        int[] array = delSurplus(a, count);
        StdOut.print("去重后结果为：");
        for (int i = 0; i < array.length; i++) {
            StdOut.print("array["+i+"]="+array[i]+" ");
        }
        StdOut.println();

    }
}

public class Practise_01_01_28 {
    public static int removeDuplicates(int[] a) {
        if (a.length == 0) return 0;
        int j = 0;
        for (int i = 0; i < a.length; i++) {
            if (a[j] != a[i]) {
                a[++j] = a[i];
            }
        }
        return j + 1;
    }
    public static void main(String[] args) {
        int[] a = {1, 2, 3, 4, 4, 4, 5, 8, 9, 10};
        int index = removeDuplicates(a);
        for (int i = 0; i < index; i++) {
            System.out.println(a[i]);
        }
    }
}
