1.2.11 根据Date的API实现一个SmartDate类型，在日期非法时抛出一个异常。

1.2.11 Develop an implementation SmartDate of our Date API that raises an exception if the date is not legal.

书上的Date的API图如下所示：

package top.leafii.practice.charpter01;

import edu.princeton.cs.algs4.StdOut;

import java.util.Objects;

public class SmartDate {
    private int month;
    private int day;
    private int year;

    public SmartDate(int month, int day, int year) {
        if(year<=0||month<=0||day<=0){
            //System.out.println("日期year month day均不可以小于等于0，无法创建！");
            throw new IllegalArgumentException("year or month or day is error.");
        }else {
            //判断闰年
            if((year % 4 == 0 && year % 100 != 0) || (year%400==0 && year % 3200 != 0) || year % 172800 == 0)
            {
                //System.out.print(year + "年是闰年。");
                if(month==2){
                    if(day>29){
                        //System.out.println("闰年2月不可以大于29天");
                        throw new IllegalArgumentException("day is error.");
                    }
                }
                //4 6 9 11
                if(month == 4 || month == 6 || month == 9 || month == 11){
                    if(day>=31){
                        //System.out.println("4，6，9，11月不可以超过30天哦");
                        throw new IllegalArgumentException("day is error.");

                    }
                }else{
//                    System.out.println("不可以超过31天哦");
                    throw new IllegalArgumentException("day is error.");
                }
            }
            else{
                //System.out.print(year + "年不是闰年。");
                if(month==2){
                    if(day>28){
                        //System.out.println("平年2月不可以大于28天");
                        throw new IllegalArgumentException("day is error.");

                    }
                }
                //4 6 9 11
                if(month == 4 || month == 6 || month == 9 || month == 11){
                    if(day>=31){
                        //System.out.println("4，6，9，11月不可以超过30天哦");
                        throw new IllegalArgumentException("day is error.");
                    }
                }else{
                    //System.out.println("不可以超过31天哦");
                    throw new IllegalArgumentException("day is error.");
                }
            }
            this.month = month;
            this.year = year;
            this.day = day;
        }
    }
    public SmartDate(String str) {

    }
    public int month(){
        return month;
    }
    public int year(){
        return year;
    }
    public int day(){
        return day;
    }

    @Override
    public String toString() {
        return "SmartDate{" +
                "month=" + month +
                ", day=" + day +
                ", year=" + year +
                '}';
    }


    public boolean equals(SmartDate that) {
        if (this == that){ return true;}
        if (that == null || getClass() != that.getClass()){ return false;}
        SmartDate smartDate = (SmartDate) that;
        return month == smartDate.month && day == smartDate.day && year == smartDate.year;
    }

    @Override
    public int hashCode() {
        return Objects.hash(month, day, year);
    }
    public static void main(String[] args)
    {
        SmartDate d=new SmartDate(9,18,2016);
        StdOut.printf("Date is:%s",d.toString());
    }
}
